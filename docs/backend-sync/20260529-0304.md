# Backend Sync — 20260529-0304

**来源 commit**: `a4528ff`

**摘要**: 后端新增 WhileLoop 全流程测试脚本，要求前端调用时必须注入 tenant_id 以支持 Lumen A2A 路由

## 前端影响
后端新增的测试脚本显示：调用 Lumen Agent 时必须传递 tenant_id (e6437035-031d-4106-affd-acffe910e9ec) 用于路由到 A2A 通道。当前前端 lg-proxy 代理未注入 tenant_id，会导致 Lumen 多轮对话中 @Polly/@Sega/@Eva 等 Agent 调用失败。

## 需要改动的文件
### `app/api/lg-proxy/[...path]/route.ts`
前端调用 LangGraph Lumen Agent 时需要注入 tenant_id 到请求 body/metadata 中，以支持后端 A2A 路由

```
// 在 proxyRequest 函数中，POST/PATCH 请求时注入 tenant_id
async function proxyRequest(
  req: NextRequest,
  method: string,
  params: { path: string[] }
) {
  const pathSegments = params.path ?? []
  const path = '/' + pathSegments.join('/')

  if (!isAllowed(path)) {
    return new Response('Forbidden', { status: 403 })
  }

  const url = new URL(path, LG_URL)
  // 透传 query params
  req.nextUrl.searchParams.forEach((v, k) => url.searchParams.set(k, v))

  const headers = new Headers()
  headers.set('X-API-Key', LG_KEY)
  req.headers.forEach((v, k) => {
    if (!['host', 'connection', 'authorization'].includes(k.toLowerCase())) {
      headers.set(k, v)
    }
  })

  let body = null
  if (method !== 'GET' && method !== 'HEAD') {
    const originalBody = await req.json().catch(() => null)
    // 注入 tenant_id 到 metadata 或 config
    if (originalBody) {
      body = JSON.stringify({
        ...originalBody,
        metadata: {
          ...(originalBody.metadata || {}),
          tenant_id: 'e6437035-031d-4106-affd-acffe910e9ec'
        },
        config: {
          ...(originalBody.config || {}),
          configurable: {
            ...(originalBody.config?.configurable || {}),
            tenant_id: 'e6437035-031d-4106-affd-acffe910e9ec'
          }
        }
      })
      headers.set('Content-Type', 'application/json')
    }
  }

  const upstreamRes = await fetch(url, { method, headers, body })
  
  // SSE stream 透传
  const contentType = upstreamRes.headers.get('content-type') || ''
  if (contentType.includes('text/event-stream')) {
    return new Response(upstreamRes.body, {
      status: upstreamRes.status,
      headers: {
        'Content-Type': 'text/event-stream',
        'Cache-Control': 'no-cache',
        'Connection': 'keep-alive'
      }
    })
  }

  return new Response(upstreamRes.body, {
    status: upstreamRes.status,
    headers: upstreamRes.headers
  })
}

export async function GET(req: NextRequest, { params }: { params: { path: string[] } }) {
  return proxyRequest(req, 'GET', params)
}

export async function POST(req: NextRequest, { params }: { params: { path: string[] } }) {
  return proxyRequest(req, 'POST', params)
}

export async function PATCH(req: NextRequest, { params }: { params: { path: string[] } }) {
  return proxyRequest(req, 'PATCH', params)
}

export async function DELETE(req: NextRequest, { params }: { params: { path: string[] } }) {
  return proxyRequest(req, 'DELETE', params)
}
```

const LG_KEY = process.env.LANGSMITH_API_KEY ?? ''
const TENANT_ID = process.env.NEXT_PUBLIC_TENANT_ID ?? 'e6437035-031d-4106-affd-acffe910e9ec'

// ... 在注入代码中使用 TENANT_ID 变量
metadata: {
  ...(originalBody.metadata || {}),

},

  ...(originalBody.config || {}),
  configurable: {
    ...(originalBody.config?.configurable || {}),

}
