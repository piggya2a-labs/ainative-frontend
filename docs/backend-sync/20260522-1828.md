```
// 新增或修改为：
export async function getTelegramBindStatus(tenantId: string) {
  const { data, error } = await supabase
    .from('tenants')
    .select('telegram_chat_id, telegram_username, telegram_bound_at, metadata')
    .eq('id', tenantId)
    .single()
  
  if (error) throw error
  
  return {
    isBound: !!data.telegram_chat_id,
    chatId: data.telegram_chat_id,
    username: data.telegram_username,
    boundAt: data.telegram_bound_at,
    bindToken: data.metadata?.telegram_bind_token,
    bindTokenExpiresAt: data.metadata?.telegram_bind_token_expires_at
  }
}
```
