// 新增：查询用户所有活跃看板的工具函数
export async function getUserActiveTenants(userId: string, supabase: SupabaseClient) {
  const { data, error } = await supabase
    .from('tenants')
    .select('slug, name, metadata, created_at')
    .eq('user_id', userId)
    .neq('status', 'archived')
    .order('created_at', { ascending: false })
  
  if (error) throw error
  return data
}

// 确保 share_token 在创建看板时生成
export function generateShareToken(tenantSlug: string): string {
  const year = new Date().getFullYear()
  const quarter = Math.ceil((new Date().getMonth() + 1) / 3)
  return `${tenantSlug}-live-${year}q${quarter}`
}
